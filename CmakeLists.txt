cmake_minimum_required(VERSION 3.27)
project(Ghoul-Engine DESCRIPTION "A Basic Game Engine" LANGUAGES CXX HOMEPAGE_URL "https://github.com/ShoweryCellar34/Ghoul-Engine")
add_executable(Ghoul-Engine "main.cpp")
add_subdirectory(Pentagram)
target_include_directories(Ghoul-Engine PUBLIC Pentagram)
target_link_libraries(Ghoul-Engine PUBLIC Pentagram)

add_custom_target(copy_resources ALL COMMAND ${CMAKE_COMMAND} -E copy_directory ${PROJECT_SOURCE_DIR}/tests/res ${PROJECT_BINARY_DIR}/res COMMENT "Copying resources into binary directory")
